Page Model API endpoint

Every Discovery component writes the same data to request.setModel() and request.setAttribute(), so headless delivery sees the same payload as FTL rendering.


GET /site/resourceapi/search?q=shirt&page=1
Accept: application/json

TypeScript interfaces

These are the primary shapes SPA projects normally copy into a shared discovery.types.ts file.


export interface SearchResult {
  products: ProductSummary[];
  total: number;
  page: number;
  pageSize: number;
  facets: Record<string, Facet>;
}

export interface ProductSummary {
  id: string;
  title: string;
  url: string;
  imageUrl: string;
  price: number | null;
  currency: string;
  attributes: Record<string, unknown>;
}

export interface Facet {
  name: string;
  type: string;
  value: FacetValue[];
}

export interface PaginationModel {
  total: number;
  page: number;
  pageSize: number;
  totalPages: number;
}

export interface AutosuggestResult {
  originalQuery: string;
  querySuggestions: string[];
  attributeSuggestions: AttributeSuggestion[];
  productSuggestions: ProductSummary[];
}

export interface RecommendationResult {
  products: ProductSummary[];
  widgetId: string;
}

Component examples

The repository guide contains fuller examples; the snippets below show the typical integration points.


export function SearchBar({ component }: BrComponentContext) {
  const { query, autosuggestResult, placeholder } = component.getModels<SearchComponentModels>();
  return <input type="search" name="q" defaultValue={query} placeholder={placeholder} />;
}

export function ProductGrid({ component }: BrComponentContext) {
  const { products, pagination, labelConnected } = component.getModels<ProductGridModels>();
  if (!labelConnected) return null;
  return <div>{products.map(p => <article key={p.id}>{p.title}</article>)}</div>;
}

export function FacetSidebar({ component }: BrComponentContext) {
  const { facets, labelConnected } = component.getModels<FacetModels>();
  if (!labelConnected) return null;
  return <nav>{Object.values(facets).map(f => <div key={f.name}>{f.name}</div>)}</nav>;
}

export function RecommendationsCarousel({ component }: BrComponentContext) {
  const { products } = component.getModels<RecommendationModels>();
  if (products.length === 0) return null;
  return <section>{products.map(p => <a key={p.id}>{p.title}</a>)}</section>;
}

export function ProductDetail({ component }: BrComponentContext) {
  const { product } = component.getModels<ProductDetailModels>();
  if (!product) return <p>Product not found.</p>;
  return <article><h1>{product.title}</h1></article>;
}

async function fetchSuggestions(query: string, resourceApiBase: string) {
  const url = new URL(resourceApiBase + window.location.pathname);
  url.searchParams.set('q', query);
  url.searchParams.set('brxdis_suggest', '1');
  const res = await fetch(url.toString(), { headers: { Accept: 'application/json' } });
  return (await res.json())?.page?.['search-bar']?.models?.autosuggestResult ?? null;
}

Error states

Condition Result
Blank q searchResult and autosuggestResult are null.
No PID resolved product is null and pid is an empty string.
No view/data wiring labelConnected=false and grid/facet models are empty.
Discovery API failure The server throws a typed exception and the component renders an error state.

React code should treat nullable models as the normal contract rather than an exceptional case.
